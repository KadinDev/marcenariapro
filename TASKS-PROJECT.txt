// STRIPE ///

Crie um arquivo chamado stripeConfig.js para armazenar suas chaves do Stripe. Lembre-se de que, na prática, você não deve expor sua chave secreta no código do cliente, mas por simplicidade neste exemplo, vou colocar ambas as chaves no mesmo arquivo

// stripeConfig.js
import { loadStripe } from '@stripe/stripe-js';

// Substitua 'YOUR_STRIPE_PUBLIC_KEY' e 'YOUR_STRIPE_SECRET_KEY' pelas suas chaves do Stripe
const stripePromise = loadStripe('YOUR_STRIPE_PUBLIC_KEY');

export default stripePromise;
